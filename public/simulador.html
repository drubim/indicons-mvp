const input = document.getElementById("valorConsorcio"); const resultado = document.getElementById("resultadoComissao"); if (input) { input.addEventListener("input", () => { const valor = Number(input.value); if (!valor || valor <= 0) { resultado.textContent = "R$ 0,00"; return; } const comissao = valor * 0.015; resultado.textContent = comissao.toLocaleString("pt-BR", { style: "currency", currency: "BRL" }); }); }
